theme_set(theme_bw())
producao <- read.csv("producao.csv", header = T)
producao <- producao |>
  mutate(data = yearmonth(as.Date(paste(data,"01"), format = "%Y-%m %d"))) |>
  pivot_longer(cols = Automoveis:Urbano, 
               names_to = "Tipo", 
               values_to = "Qtd") |>
  as_tsibble(index = data, 
             key = Tipo)
producao <- producao |>
  mutate(Segmento = case_when(
    Tipo %in% c("Automoveis", "Comerciaisleves") ~ "Veiculos",
    Tipo %in% c("Semileve", "Leve", "Medio", "Semipesado", "Pesado") ~ "Caminhoes",
    Tipo %in% c("Rodoviario", "Urbano") ~ "Onibus"
  ))
prod_plot <- producao |>
  mutate(Tipo = recode_factor(Tipo, "Rodoviario" = "Highway", "Urbano" = "Urban", "Comerciaisleves" = "Light comercial", "Automoveis" = "Passenger cars",  "Leve" = "Light", "Semileve" = "Medium light", "Medio" = "Medium", "Semipesado" = "Medium heavy", "Pesado" = "Heavy")) |>
  mutate(Segmento = recode_factor(Segmento, "Onibus" = "Buses", "Veiculos" = "Cars", "Caminhoes" = "Trucks")) |>
  mutate(across(!c(data, Qtd), as.factor))

prod_plot |>
  autoplot(Qtd, aes(color = Tipo)) +
  facet_wrap(~ Segmento, scales = "free_y", nrow=3) + 
  theme(legend.title = element_blank()) +
  # theme(legend.position = "bottom") +
  labs(x="date [1M]", y="Qtt")
producao |>
  group_by(Segmento) |>
  summarise(Qtd = sum(Qtd, na.rm = T)) |>
  autoplot(Qtd, aes(color = Segmento)) +
  theme(legend.title = element_blank()) +
  theme(legend.position = "bottom")
producao |>
  summarise(Qtd = sum(Qtd, na.rm = T)) |>
  autoplot(Qtd)
producao_hts <- producao |>
  aggregate_key(Segmento/Tipo, Qtd = sum(Qtd))

producao_hts <- producao_hts |>
 mutate(parada_covid = if_else(data >= yearmonth("2020-04") & 
                                 data <= yearmonth("2020-06"), 
                                 1, 0))
all_dates <- sort(unique(producao_hts$data))
N <- length(all_dates)

path <- "Resultados_reconc"
if(!dir.exists(path)) dir.create(path)
arquivo_csv1 <- file.path(path, "metricas_agg.csv")
arquivo_csv2 <- file.path(path, "metrics_nagg_segmento.csv")
arquivo_csv3 <- file.path(path, "metrics_nagg.csv")

w <- 12 # windows size of rolling forecasting origin

for (h in 1) { ### ATENTION: In the paper we did h=1:6 ###
  
  message("=== Horizonte h = ", h, " ===")
  
  for (i in (N-w-h+1):(N - h)) {
    
    cut_date <- all_dates[i]
    
    message("i = ", i, " | cut_date = ", cut_date)
    
    train_set <- producao_hts |> 
      fill_gaps() |>
      filter(data <= cut_date)
    
    test_dates <- all_dates[(i+1):(i+h)]
    
    test_set <- producao_hts |> 
      fill_gaps() |>
      filter(data %in% test_dates)
  
fit_arima <- train_set |>
  model(base = ARIMA(Qtd ~ parada_covid
                     )) |>
  reconcile(
    bu = bottom_up(base),
    ols = min_trace(base, method = "ols"),
    mint = min_trace(base, method = "mint_shrink"),
    wlsv = min_trace(base, method = "wls_var"),
    wlss = min_trace(base, method = "wls_struct") #,
    # mintc = min_trace(base, method = "mint_cov")
  )

assign(paste0("fit_arima_h", h, "_i", i), fit_arima)


fc_arima <- fit_arima |> 
  forecast(new_data = test_set)

assign(paste0("fc_arima_h", h, "_i", i), fc_arima)

metrics_agg <- fc_arima |>
  filter(is_aggregated(Tipo), is_aggregated(Segmento)) |>
  accuracy(
    data = producao_hts,
    measures = list(rmse = RMSE, mae = MAE, mape = MAPE)
  ) |>
  group_by(.model) |>
  summarise(armse = mean(rmse, na.rm = TRUE), amae = mean(mae, na.rm = TRUE), amape = mean(mape, na.rm = TRUE),
            srmse = sd(rmse, na.rm = TRUE), smae = sd(mae, na.rm = TRUE), smape = sd(mape, na.rm = TRUE)) |>
        mutate(h = h, i = i) 

      write.table(
        metrics_agg,
        file = arquivo_csv1,
        append = TRUE,
        sep = ",",
        row.names = FALSE,
        col.names = !file.exists(arquivo_csv1)
        )
